JavaScript - while

  • 1. while 문은 반복문입니다.
  • 2. 조건식이 참인 동안 코드 블록을 반복 실행합니다.
  • 3. 조건식이 거짓이면 코드 블록을 실행하지 않습니다.
  • 4. 조건식이 참인 동안 코드 블록을 반복 실행합니다.

while 실행 전 코드

<script> // 사용자로부터 최대 숫자 입력 받기 var input = prompt("최대 숫자를 입력해주세요:"); var maxNum = parseInt(input); // 입력받은 값을 정수로 변환 // 입력값 검증 if (isNaN(maxNum) || maxNum < 1) { document.write("1 이상의 유효한 숫자를 입력해주세요."); } else { var num = 1; // 시작 숫자 while (num <= maxNum) { document.write("num은 " + num + "입니다.<br>"); num++; } } </script>
