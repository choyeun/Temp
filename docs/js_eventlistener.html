javascript - EventListener

  • 1. EventListener는 이벤트를 처리하는 객체입니다.
  • 2. EventListener는 이벤트를 추가하거나 제거할 수 있습니다.
  • 3. EventListener는 이벤트를 처리하는 함수를 지정합니다.
  • 4. EventListener는 이벤트를 처리하는 객체를 지정합니다.

EventListener 실행 전 코드

<script> var feedback = document.getElementById('feedback'); // 클릭 이벤트 리스너 document.getElementById('clickButton').addEventListener('click', function() { feedback.innerHTML = '버튼이 클릭되었습니다!'; feedback.style.color = 'green'; }); // 마우스 오버 이벤트 리스너 document.getElementById('hoverButton').addEventListener('mouseover', function() { feedback.innerHTML = '마우스가 버튼 위에 있습니다!'; feedback.style.color = 'blue'; }); // 키보드 이벤트 리스너 document.getElementById('inputBox').addEventListener('keyup', function(event) { feedback.innerHTML = '키가 입력되었습니다: ' + event.key; feedback.style.color = 'red'; }); </script>
