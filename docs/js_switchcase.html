JavaScript - switch case

  • 1. switch case 문은 다중 조건문입니다.
  • 2. switch case 문은 조건식을 평가하여 일치하는 case 문을 실행합니다.
  • 3. case 문이 일치하지 않으면 default 문을 실행합니다.
  • 4. default 문은 생략할 수 있습니다.

switch case 실행 전 코드

<script> // 사용자로부터 숫자 입력 받기 var input = prompt("숫자를 입력해주세요:"); var num = parseInt(input); // 입력받은 값을 정수로 변환 // 입력값 검증 if (isNaN(num)) { document.write("유효한 숫자를 입력해주세요."); } else { switch (num) { case 1: document.write("입력한 숫자는 1입니다."); break; case 2: document.write("입력한 숫자는 2입니다."); break; case 3: document.write("입력한 숫자는 3입니다."); break; default: document.write("입력한 숫자는 1, 2, 3이 아닙니다."); } } </script>
